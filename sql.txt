CREATE TABLE aeonShopping.tnews (
  `NEWS_NO` varchar(20) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT 'NEWS NUMBER ID',
  `NEWS_NAME` varchar(500) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT 'NEWS NAME',
  `TYPE_NEWS` varchar(10) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT 'TYPE NEWS (TCODE T0002)',
  `CREATE_DATE` timestamp NULL DEFAULT NULL COMMENT 'CREATE DATE',
  `INSERT_ID` varchar(20) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT 'INSERT Id',
  `MODIFY_DATE` timestamp NULL DEFAULT NULL COMMENT 'MODIFY DATE',
  `MODIFY_ID` varchar(20) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT 'MODIFY ID',
  PRIMARY KEY (`NEWS_NO`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

  `MODIFY_DATE` timestamp NULL DEFAULT NULL COMMENT 'MODIFY DATE',
  `MODIFY_ID` varchar(20) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT 'MODIFY ID',
